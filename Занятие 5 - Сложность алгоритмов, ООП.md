4. Создать класс PersonThree c get и set для поля name и конструктором, сделать класс наследник от класса Person.

```

class PersonThree {
  name;
  constructor(name) {
    this.name = name;
  }

  get name() {
    return this.name;
  }

  set name(value) {
    this.name = value;
  }
}

class PersonThreeSuccessor extends PersonThree {
```
